class SyntheticPriceEngine {
  constructor(startPrice = 1850.50) {
    this.price = startPrice;
    this.history = [];
    this.maxHistory = 100;
    this.trend = 0;
    this.volatility = 0.0003;
    this.drift = 0;
    this.regimeSwitchProbability = 0.02;
    this.minVolatility = 0.0001;
    this.maxVolatility = 0.0006;
  }

  switchRegime() {
    if (Math.random() < this.regimeSwitchProbability) {
      this.trend = [-1, 0, 1][Math.floor(Math.random() * 3)];
      this.volatility = this.minVolatility + Math.random() * (this.maxVolatility - this.minVolatility);
    }
  }

  generateTick(userBias = 0) {
    this.switchRegime();
    const momentum = this.trend * (0.00005 + Math.random() * 0.00015);
    const noise = this.boxMullerGaussian() * this.volatility;
    const bias = userBias * 0.00005;
    const delta = momentum + noise + bias;
    this.price += delta;
    if (this.price < 0.01) this.price = 0.01;
    this.history.push(this.price);
    if (this.history.length > this.maxHistory) this.history.shift();
    return parseFloat(this.price.toFixed(5));
  }

  boxMullerGaussian() {
    const u1 = Math.random() || 0.0001;
    const u2 = Math.random();
    return Math.sqrt(-2 * Math.log(u1)) * Math.cos(2 * Math.PI * u2);
  }

  getCurrentPrice() { return this.price; }
  getHistory() { return [...this.history]; }
  reset(startPrice) {
    this.price = startPrice;
    this.history = [];
    this.trend = 0;
    this.volatility = 0.0003;
  }
}

export default SyntheticPriceEngine;

class CandleBuilder {
  constructor() {
    this.reset();
  }

  reset() {
    this.open = null;
    this.high = -Infinity;
    this.low = Infinity;
    this.close = null;
  }

  update(price) {
    if (this.open === null) {
      this.open = price;
    }
    this.high = Math.max(this.high, price);
    this.low = Math.min(this.low, price);
    this.close = price;
  }

  getCandle(timestamp) {
    const candle = {
      time: timestamp,
      open: this.open,
      high: this.high,
      low: this.low,
      close: this.close
    };
    this.reset();
    return candle;
  }

  hasData() {
    return this.open !== null;
  }
}

export default CandleBuilder;

export function calculateUserBias(activeTrades) {
  if (!activeTrades || activeTrades.length === 0) return 0;
  
  const buys = activeTrades.filter(t => t.direction === 'buy').length;
  const sells = activeTrades.filter(t => t.direction === 'sell').length;
  
  if (buys > sells) {
    return -1;  // Slight downward pressure
  } else if (sells > buys) {
    return 1;   // Slight upward pressure
  }
  
  return 0;
}

export function calculateHouseEdge(tradeAmount, basePayoutPercentage = 85) {
  // Reduce payout slightly based on amount
  const amountFactor = Math.min(tradeAmount / 1000, 0.05); // Max 5% reduction
  const adjustedPayout = basePayoutPercentage * (1 - amountFactor);
  
  return {
    payoutPercentage: Math.max(75, Math.floor(adjustedPayout)),
    houseEdgePercentage: 100 - Math.max(75, Math.floor(adjustedPayout))
  };
}

      width: chartContainerRef.current.clientWidth,
      height: 400,
      layout: {
        background: { color: '#0f172a' },
        textColor: '#d1d5db',
      },
      grid: {
        vertLines: { color: '#1f2937' },
        horzLines: { color: '#1f2937' },
      },
      crosshair: {
        mode: 1,
      },
      rightPriceScale: {
        borderColor: '#374151',
      },
      timeScale: {
        borderColor: '#374151',
        timeVisible: true,
        secondsVisible: false,

      downColor: '#ef4444',
      borderUpColor: '#22c55e',
      borderDownColor: '#ef4444',
      wickUpColor: '#22c55e',
      wickDownColor: '#ef4444',

    candleSeriesRef.current = candleSeries;
