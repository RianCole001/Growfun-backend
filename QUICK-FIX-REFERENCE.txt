2. CHART BECOMES DORMANT
   
   Problem: Chart stops updating after placing trade
   
   Cause: Price polling interval gets cleared
   
   Solution: Use independent polling with empty dependency array
   
   useEffect(() => {
     const fetchPrices = async () => {
       const res = await axios.get('/api/binary/prices/');
       setPrices(res.data.prices);
     };
     
     fetchPrices();
     const interval = setInterval(fetchPrices, 1000);
     
     return () => clearInterval(interval);
   }, []); // ← Empty array is critical!

================================================================================
COMMON MISTAKES
================================================================================

❌ Don't do this:
   useEffect(() => {
     // polling code
   }, [trades, balance, mode]); // ← Dependencies cause re-runs

✅ Do this:
   useEffect(() => {
     // polling code
   }, []); // ← Empty array, runs once

❌ Don't do this:
   if (error) {
     clearInterval(priceInterval); // ← Stops polling
   }

✅ Do this:
   if (error) {
     console.error(error); // ← Log but keep polling
   }
